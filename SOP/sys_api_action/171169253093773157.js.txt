(function (request, response) {
  const reqBody = request.getBody();
  const tableName = request.getQueryParams().table_name;
  
  const records = new SimpleRecord(tableName);
  records.addEncodedQuery(reqBody.condition);
  records.query();
  const result = [];
  while (records.next()) {
    result.push(records.getValue('sys_id'));
  }
  response.setBody({
    count: result.length,
    result: result,
  });
})(SimpleApiRequest, SimpleApiResponse);
